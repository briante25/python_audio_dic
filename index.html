3.寫入程式碼

import speech_recognition as sr
r = sr.Recognizer()

with sr.Microphone() as source:
    print('請開始說話')
    audio = r.listen(source) #接收值
    
    try:
        text = r.recognize_google(audio,language="zh-TW") #將接收值以中文型態接收
        print('正在轉換成文字...')
        print(text) #輸出語音文字
    except:
         print('很抱歉，請再說一次') #無收到音訊則會顯示此列

5.導入網頁爬蟲必要套件

import speech_recognition as sr
import requests
r = sr.Recognizer()

with sr.Microphone() as source:
    print('請開始說話')
    audio = r.listen(source)
    
    try:
        text = r.recognize_google(audio,language="zh-TW")
        print('正在轉換成文字...')
        url=f"https://pedia.cloud.edu.tw/Home/List?search={text}" #串在後面
        html=requests.get(url)
        html.encoding="utf-8"
        print(html.text)#輸出值
    except Exception as e:
        print(e)

針對網頁原始碼去解析

我這邊直接把html顯示出來，為了方便改成到線上Google Colaboratory去解析。

可以看到目標被包在div內部，並且我們找到了class的名稱，那就直接抓出來

看起來是成功了，那我們接著改到Anaconda上去執行，順便指定第[3]個，因為他是成語辭典的索引值。
import speech_recognition as sr
import requests
from bs4 import BeautifulSoup
r = sr.Recognizer()

with sr.Microphone() as source:
    print('請開始說話')
    audio = r.listen(source)
    
    try:
        text = r.recognize_google(audio,language="zh-TW")
        print('正在轉換成文字...')
        print(text)
        url=f"https://pedia.cloud.edu.tw/Home/List?search={text}"
        html=requests.get(url)
        html.encoding="utf-8"
       
        sp=BeautifulSoup(html.text,"html.parser")
        data=sp.find_all("div",class_="wordDefination")[3]
        print(data.text)
    except Exception as e:
        print(e)
